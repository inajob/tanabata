ブックマークレットなので、上記リンクをブックマークバーにD&Dすればいつでも願えます。 

javascript:
  tanzaku=prompt('ねがいごと');
  out='┌┴┐\n';
  for(var i = 0; i != tanzaku.length; i ++){
    out += '│' + tanzaku.charAt(i) + '│\n'
  };
  out += '└─┘\n#tanabata #bookmarklet http://inajob.github.io/tanabata/';
  document.location.href=('https://twitter.com/intent' + '/tweet?text=' + encodeURIComponent(out));
